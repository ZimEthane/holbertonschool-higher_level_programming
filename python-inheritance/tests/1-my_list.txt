import :
    >>> MyList = __import__('1-my_list').MyList

create empty list :
    >>> my_list = MyList()
    >>> print(my_list)
    []

add elements :
    >>> my_list.append(1)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.append(3)
    >>> my_list.append(5)
    >>> print(my_list)
    [1, 4, 2, 3, 5]

print sorted list :
    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]

check original list not modified :
    >>> print(my_list)
    [1, 4, 2, 3, 5]

list with one element :
    >>> one = MyList()
    >>> one.append(42)
    >>> one.print_sorted()
    [42]

empty list print_sorted :
    >>> empty = MyList()
    >>> empty.print_sorted()
    []
